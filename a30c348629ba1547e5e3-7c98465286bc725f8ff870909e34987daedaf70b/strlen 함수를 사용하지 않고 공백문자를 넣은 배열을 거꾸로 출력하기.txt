#include "stdio.h"
#include "string.h"

int main()
{
	char a[100];
	printf("입력 :");
	scanf("%[^\n]",a);// %[] 는 []안에 있는 문자들을 받는다는것! 앞에 ^를 붙이면 그문자를 제외한 모든 문자를 받는다는것이다

	int i=0;
	int len=-1;
	while(1)
	{
		if(a[i] == '\0')
			break;
		len++;
		i++;
	}
	for(i=len;i>=0;i--)
	{
		printf("%c",a[i]);
	}
	getchar();
	return 0;
}
